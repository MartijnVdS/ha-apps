#include <tunables/global>

profile pv2mqtt flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  file,
  signal (send) set=(kill,term,int,hup,cont),

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Access to options.json and other files within your addon
  /data/** rw,

  # Start new profile for service
  /usr/local/bin/pv2mqtt cx -> pv2mqtt_app,

  profile pv2mqtt_app flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>

    # Receive signals from S6-Overlay
    signal (receive) peer=*_pv2mqtt,

    # Access to options.json and other files within your addon
    /data/** rw,

    # Access required for service functionality
    /usr/local/bin/pv2mqtt r,
    /bin/bash rix,
    /bin/echo ix,
    /etc/passwd r,
    /dev/tty rw,

    # For MQTT and Modbus-TCP
    network inet stream,
    network inet6 stream,

    # If Modbus RTU (Serial) is used, we need access to serial devices
    # Home Assistant addons typically map these to /dev/tty*
    /dev/tty[A-Za-z]* rw,
    /dev/serial/by-id/* rw,
  }
}
